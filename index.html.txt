<script>
let name = "";
let qIndex = 0;

const questions = [
{
q: "If someone stares at you for 5 seconds, what do you do?",
a: ["Stare back 😎","Smile softly 😊","Ignore like a queen 👑"],
c: ["That confidence... dangerous 😌","That smile? lethal 💘","Royal attitude detected 👑"]
},
{
q: "What’s your secret weapon?",
a: ["Eyes 👀","Smile 😊","Attitude 🔥"],
c: ["Those eyes... trouble 😏","That smile wins hearts 💖","Attitude? irresistible 🔥"]
},
{
q: "If I tried to prank you?",
a: ["Catch instantly 😏","Maybe 😶","Depends 😉"],
c: ["Too smart huh 😌","Interesting... risky 😏","Now that’s intriguing 😈"]
},
{
q: "Pick your vibe",
a: ["Cute 🐣","Bold 🔥","Mystery 🌙"],
c: ["Cute but powerful 💕","Bold & stunning 😌","Mysterious charm ✨"]
},
{
q: "When complimented?",
a: ["Blush ☺️","Say thanks 😌","Tease 😜"],
c: ["That blush says everything 💖","Confidence looks good on you 😏","Playful huh 😜"]
}
];

function start(){
document.getElementById("welcome").classList.add("hidden");
document.getElementById("nameScreen").classList.remove("hidden");
}

function saveName(){
name = document.getElementById("nameInput").value;
if(name==="") return;

document.getElementById("nameMsg").innerText =
`Hmm… ${name}, your name is as beautiful as you are, cutie 💕`;

setTimeout(()=>{
document.getElementById("nameScreen").classList.add("hidden");
document.getElementById("questionScreen").classList.remove("hidden");
showQuestion();
},1500);
}

function showQuestion(){
let q = questions[qIndex];
document.getElementById("question").innerText = q.q;
let optionsHTML = "";

q.a.forEach((opt,i)=>{
optionsHTML += `<button onclick="answer(${i})">${opt}</button><br>`;
});

document.getElementById("options").innerHTML = optionsHTML;
document.getElementById("compliment").innerText = "";
}

function answer(i){
let q = questions[qIndex];
document.getElementById("compliment").innerText = q.c[i];

setTimeout(()=>{
qIndex++;
if(qIndex < questions.length){
showQuestion();
}else{
showLoading();
}
},1200);
}

function showLoading(){
document.getElementById("questionScreen").classList.add("hidden");
document.getElementById("loading").classList.remove("hidden");

let steps = [
"Scanning intelligence…",
"Measuring charm levels…",
"Checking prank resistance…"
];

let i=0;
let width=0;

let interval = setInterval(()=>{
document.getElementById("loadingText").innerText = steps[i%3];
width += 20;
document.getElementById("bar").style.width = width+"%";
i++;

if(width>=100){
clearInterval(interval);
showResult();
}
},700);
}

function showResult(){
document.getElementById("loading").classList.add("hidden");
document.getElementById("result").classList.remove("hidden");

let text = `${name}, you are incredibly smart, charming, and almost impossible to prank… 😌 Honestly, I was impressed the whole time…`;

document.getElementById("resultText").innerText = text;

setTimeout(()=>{
document.getElementById("prank").classList.remove("hidden");
document.getElementById("finalLine").innerText =
"Don’t worry… even smart and beautiful girls like you fall for the best pranks 😉💖";
},2000);
}

function restart(){
location.reload();
}
